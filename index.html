from flask import Flask, request, render_template import os import secrets app = Flask(__name__) UPLOAD_FOLDER = 'uploads' # Create this folder app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER ALLOWED_EXTENSIONS = {'txt', 'pdf', 'png', 'jpg', 'jpeg', 'gif'} # Limit allowed extensions def allowed_file(filename): return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS @app.route('/', methods=['GET', 'POST']) def upload_file(): if request.method == 'POST': # check if the post request has the file part if 'file' not in request.files: return 'No file part' file = request.files['file'] # If the user does not select a file, the browser submits an # empty file without a filename. if file.filename == '': return 'No selected file' if file and allowed_file(file.filename): extension = file.filename.rsplit('.', 1)[1].lower() random_filename = secrets.token_hex(16) + '.' + extension # Generate a random filename file.save(os.path.join(app.config['UPLOAD_FOLDER'], random_filename)) return 'File successfully uploaded!' return render_template('upload_form.html') if __name__ == '__main__': app.run(debug=True)
